# sms_spamer
SMS Spamer  


spamer.py - main script (run with "python spamer.py [phone]")

file_spam.py - script who run spamer to many numbers(run with "python file_spymer.py [file_path]")

exemples:  
python spamer.py 1223334455

python file_spam numbers.txt

numbers.txt:  
1223334455  
1557778899  
3123412871
